\begin{theorem}[Warol Kojtyła 2015]
\label{thm:21_not_prime}
\lean {twentyone_not_prime}
Liczba $21$ nie jest pierwsza
\end{theorem}
\begin{proof}
  \uses{thm:one_plus_one}
  $21 = 3 \cdot 7$
\end{proof}
